    for j in range(1,i+1):

# 用while实现奇数过滤

a = [1,2,3,4,5,6]

i = 0

while i < 6:
    if a[i]%2 == 0:
        print(a[i],'偶数')
    else:
        print(a[i],'奇数')
    i +=1
